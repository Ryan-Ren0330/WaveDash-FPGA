Dynamical Systems¶

The trial-averaged firing rate for all 272 neurons between -500 ms and +500 ms from word onset (in 200 time bins of width 5 ms, smoothed with a Gaussian kernel of 50 ms width) is given in Fig4-planning_production_dynamics.xlsx. From these data, subspaces are constructed based on the principal components of ensemble activity during the planning period (-500 ms to word onset) and production (word onset to +500 ms). Metrics of subspace (mis)alignment are calculated in comparison with the null distribution expected for random subspace partitions.

def subspace_error(U, U_hat):
    """Calculates subspace projection error between two orthogonal subspaces."""
    I = np.identity(U.shape[0])
    top = np.linalg.norm((I - U_hat @ U_hat.T) @ U, ord="fro")
    bottom = np.linalg.norm(U, ord="fro")
    error = float(top) / float(bottom)
    return error

def variance_explained(trial_average_fr, original_space, test_space):
    """Calculates the fraction of variance explained if a system is projected into a different space."""
    return np.trace(test_space.components_ @ np.cov(trial_average_fr) @ test_space.components_.T) / sum(original_space.explained_variance_)

def simulate_nulls(combined_avg_fr, N_COMPONENTS=10):
    A = combined_avg_fr.T
    M = np.mean(A.T, axis=1)
    C = A - M
    V = np.cov(C.T)
    values, vectors = np.linalg.eig(V)
    U = np.real(vectors)
    S = np.real(values)
    U = U * np.sign(S)
    S = S * np.sign(S)

    alignment = []
    error = []

    for j in tqdm(range(10000)):
        V = np.random.normal(loc=0, scale=1, size=(combined_avg_fr.shape[0], N_COMPONENTS*2))
        idx1 = np.random.choice(range(V.shape[1]), size=N_COMPONENTS)
        idx2 = np.array([i for i in range(V.shape[1]) if i not in idx1])
        Z1 = scipy.linalg.orth((U * np.sqrt(S) @ V[:,idx1]) / np.linalg.norm(U * np.sqrt(S) @ V[:,idx1]))
        Z2 = scipy.linalg.orth((U * np.sqrt(S) @ V[:,idx2]) / np.linalg.norm(U * np.sqrt(S) @ V[:,idx2]))
        alignment.append(np.trace(Z2.T @ Z1 @ Z1.T @ Z2)/N_COMPONENTS)
        error.append(subspace_error(Z1, Z2))

    true_error = subspace_error(PCA(N_COMPONENTS).fit(combined_avg_fr[:,:combined_avg_fr.shape[1]//2].T).components_.T, PCA(N_COMPONENTS).fit(combined_avg_fr[:,combined_avg_fr.shape[1]//2:].T).components_.T)
    true_alignment = subspace_error(PCA(N_COMPONENTS).fit(combined_avg_fr[:,:combined_avg_fr.shape[1]//2].T).components_.T, PCA(N_COMPONENTS).fit(combined_avg_fr[:,combined_avg_fr.shape[1]//2:].T).components_.T)
    return true_error, alignment, error
